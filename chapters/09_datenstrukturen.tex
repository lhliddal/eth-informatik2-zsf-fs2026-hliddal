\section{Datenstrukturen}\label{sec:09-datenstrukturen}
\subsection{Queue und Stack}\label{sec:09-queue-und-stack}
\begin{defbox}[Queue]
Eine Warteschlange, bei der Elemente nach dem \textit{First In – First Out} (FIFO) Prinzip verarbeitet werden.
\begin{factlist}
\ZSFFact{\texttt{add/enqueue}}{Element am Ende der Queue einfügen}
\ZSFFact{\texttt{remove/dequeue}}{Element am Anfang der Queue entfernen}
\end{factlist}
\end{defbox}

\begin{defbox}[Stack]
Ein Stapelspeicher, der nach dem \textit{Last In – First Out} (LIFO) Prinzip arbeitet.
\begin{factlist}
\ZSFFact{\texttt{add/push}}{Element oben auf den Stack legen}
\ZSFFact{\texttt{remove/pop}}{Oberstes Element entfernen}
\end{factlist}
\end{defbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
stack = []                        # Stack (LIFO)
stack.append(x)                   # push   -> O(1)
top = stack.pop()                 # pop    -> O(1)

from collections import deque     # Queue (FIFO)
q = deque()
q.append(x)                       # enqueue-> O(1)
first = q.popleft()               # dequeue-> O(1)
# list.pop(0) waere O(n) -> deque benutzen!
\end{lstlisting}
\end{codebox}

\subsection{Binary (Search) Trees}\label{sec:09-binary-search-trees}
\subsubsection{Baumdurchläufe (Traversal)}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def preorder_list(node: Optional[Node]) -> list:
    if node is None:
        return []
    return ([node.key] + preorder_list(node.left)
            + preorder_list(node.right))

def postorder_list(node: Optional[Node]) -> list:
    if node is None:
        return []
    return (postorder_list(node.left)
            + postorder_list(node.right) + [node.key])

def inorder_list(node: Optional[Node]) -> list:
    if node is None:
        return []
    return (inorder_list(node.left)
            + [node.key] + inorder_list(node.right))
\end{lstlisting}
\end{codebox}

\begin{factlist}
\ZSFFact{Preorder}{Hauptreihenfolge, mit dem wird Baum gebaut}
\ZSFFact{Postorder}{Nebenreihenfolge}

\subsubsection{Knoten suchen (search)}
\begin{statementbox}
Durchschnittsfall: $\Theta(\log n)$, Worst Case: $\Theta(n)$
\end{statementbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def search(self, key: int) -> Optional[Node]:
    """Search for the node with the specified key in the tree and return it, or None if there is no such node."""
    n = self.root  # Starte bei der Wurzel
    while n is not None and n.key is not key:  # Laufe, bis Schlüssel gefunden oder Ende erreicht
        if key < n.key:  # Gesuchter Schlüssel ist kleiner -> links weiter
            n = n.left  # Gehe zum linken Kind
        else:
            n = n.right  # Sonst nach rechts gehen
    return n  # Gefundener Knoten oder None
\end{lstlisting}
\end{codebox}

\subsubsection{Knoten hinzufügen (add)}
\begin{statementbox}
Durchschnittsfall: $\Theta(\log n)$, Worst Case: $\Theta(n)$
\end{statementbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def add(self, key: int) -> bool:
    """Add a node with the specified key to the tree.
    Return True if the node was added, and False otherwise."""
    if self.root is None:
        self.root = Node(key)
        return True
    n = self.root
    while n.key is not key:
        if key < n.key:
            if n.left is None:
                n.left = Node(key)
                return True
            n = n.left
        else:
            if n.right is None:
                n.right = Node(key)
                return True
            n = n.right
    return False
\end{lstlisting}
\end{codebox}


\subsubsection{Knoten löschen (delete)}
\begin{statementbox}
In allen Fällen: $\Theta(n)$
\end{statementbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def deleteNode(root, key):
    if root is None:
        return None
    if key < root.val:
        root.left = deleteNode(root.left, key)
    elif key > root.val:
        root.right = deleteNode(root.right, key)
    else:
        # Case 1: No children
        if root.left is None and root.right is None:
            root = None
        # Case 2: Only right child
        elif root.left is None:
            root = root.right
        # Case 2: Only left child
        elif root.right is None:
            root = root.left
        # Case 3: Two children
        else:
            successor = findSuccessor(root)
            root.val = successor.val
    return root
def findSuccessor(start_node):
    parent = start_node
    node = start_node.right
    while node.left is not None:
        parent = node
        node = node.left
    parent.left = node.right
    return node
\end{lstlisting}
\end{codebox}
\subsubsection{Komplexitäten}
\begin{contentbox}
\begin{factlist}
\ZSFFact{Baum aufbauen (Worst Case)}{$\Theta(n^2)$ \; (balanciert: $\Theta(n \log n)$)}
\ZSFFact{Minimum/Maximum finden}{$\mathcal{O}(n)$ \; (balanciert: $\mathcal{O}(\log n)$)}
\ZSFFact{Einfügen}{$\mathcal{O}(n)$ \; (balanciert: $\mathcal{O}(\log n)$)}
\ZSFFact{Entfernen (beliebiges Element)}{$\mathcal{O}(n)$ \; (balanciert: $\mathcal{O}(\log n)$)}
\end{factlist}
\end{contentbox}

\end{center}

\subsubsection{Grundidee eines Binary Heaps}
\begin{defbox}[Binary Heap]
Ein Binary Heap ist eine vollständige Binärstruktur, bei der gilt:
\begin{factlist}
\ZSFFact{Min-Heap}{Elternknoten ist stets kleiner als seine Kinder.}
\ZSFFact{Max-Heap}{Elternknoten ist stets größer als seine Kinder.}
\end{factlist}

Heaps werden in einer Liste gespeichert, sodass die Knotenebenen hintereinander notiert sind:
\[
\texttt{[1, 2, 3, 4, 5, 6]} \Rightarrow \text{Index: } 0,1,2,\dots
\]
\end{defbox}

\begin{contentbox}[Zugriffsformeln]
\begin{factlist}
\ZSFFact{\texttt{parent}}{$(i-1)//2$}
\ZSFFact{\texttt{left child}}{$2i + 1$}
\ZSFFact{\texttt{right child}}{$2i + 2$}
\ZSFFact{Höhe}{$\log_2(N+1)$ mit $N = \texttt{len(list)}$}
\end{factlist}
\end{contentbox}

\subsubsection{Heapify (Heap aufbauen)}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def heapify(list):
    for i in range(len(list)):
        while list[(i - 1)//2] > list[i] and i > 0:
            swap(list, (i - 1)//2, i)
            i = (i - 1)//2
    return
\end{lstlisting}
\end{codebox}
\begin{contentbox}
oder:
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def heapify(list):
    n = len(list)
    for i in range(n//2 - 1, -1, -1):
        shift_down(list, i, n)
    return
\end{lstlisting}
\end{codebox}

\subsubsection{SiftDown (oberstes Element löschen)}
\begin{contentbox}
Beim Entfernen der Wurzel (z.\,B. Max-Element bei Max-Heap) wird das letzte Element nach oben geschoben und mit seinen Kindern verglichen. Dabei wird immer mit dem größeren (bzw. kleineren bei Min-Heap) Kind getauscht.
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def SiftDown(a, i, m):
    while 2*i + 1 < m:
        j = 2*i + 1
        if j + 1 < m and a[j] < a[j + 1]:
            j += 1
        if a[i] >= a[j]:
            break
        a[i], a[j] = a[j], a[i]
        i = j
\end{lstlisting}
\end{codebox}

\subsubsection{Sortieren mit Heaps (Heap Sort)}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def SortHeap(a):
    heapify(a)
    for i in reversed(range(len(a))):
        swap(a[0], a[i])
        SiftDown(a, 0, i)
    return
\end{lstlisting}
\end{codebox}

\subsubsection{Elemente einfügen und löschen}
\begin{contentbox}
\begin{factlist}
\ZSFFact{Einfügen}{ans Ende der Liste hängen, dann \texttt{siftUp()}}
\ZSFFact{Löschen}{Element mit letztem tauschen, löschen, dann \texttt{SiftDown()}}
\end{factlist}
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def removeMax(heap):
    if len(heap) == 0:
        return None
    max_val = heap[0]
    heap[0] = heap[-1]
    heap.pop()
    SiftDown(heap, 0, len(heap))
    return max_val
\end{lstlisting}
\end{codebox}


\subsubsection{Komplexitäten}
\begin{contentbox}
\begin{factlist}
\ZSFFact{Heap aufbauen (Worst Case)}{$\Theta(n \log n)$}
\ZSFFact{Maximum/Minimum holen}{$\mathcal{O}(1)$}
\ZSFFact{Einfügen in Heap}{$\mathcal{O}(\log n)$}
\ZSFFact{Entfernen (Top-Element)}{$\mathcal{O}(\log n)$}
\end{factlist}
\end{contentbox}




\subsection{Hash-Tabellen}\label{sec:09-hash-tabellen}

\subsubsection{Grundidee}
\begin{defbox}[Hash-Tabelle]
Eine Hash-Tabelle speichert ungeordnete Sammlungen \textbf{einzigartiger Elemente} (z.\,B. \texttt{set} oder \texttt{dict} in Python).

\ZSFkeyword{Hashing}: Einer Eingabe wird mittels \ZSFkeyword{Hash-Funktion} ein Integer zugewiesen. Die Platzierung im Array der Länge $M$ erfolgt am Index:
\[
\texttt{index} = \texttt{hash}(x)\ \%\ M \quad(\text{math: } \bmod)
\]
\end{defbox}

\subsubsection{Laufzeiten}
\begin{contentbox}
\begin{factlist}
\ZSFFact{Suche, Einfügen, Entfernen (Erwartungswert)}{$\mathcal{O}(1)$}
\ZSFFact{Suche, Einfügen, Entfernen (Worst Case)}{$\mathcal{O}(n)$}
\end{factlist}
\end{contentbox}

\subsubsection{Kollisionsbehandlung}
\begin{contentbox}
Treffen mehrere Elemente auf denselben Index, gibt es zwei Lösungsansätze:
\begin{factlist}
\ZSFFact{Chaining}{Am Index wird der Head-Knoten einer \ZSFkeyword{Linked List} gespeichert, die alle kollidierenden Elemente aufnimmt. \ZSFref{sec:12-linked-lists}\\
  \emph{Vorteil:} Ein oft wiederholter Hash blockiert nicht die restliche Tabelle.}
\ZSFFact{Probing}{Ist der Index besetzt, wird zirkulär der nächste freie Index (\ZSFkeyword{Wrap-around}) gesucht. \\
  \emph{Vorteil:} Alle Daten bleiben Cache-freundlich im Array. \\
  \emph{Nachteil:} Löschungen erfordern \ZSFkeyword{Tombstoning} (spezielle Literale als Platzhalter) oder komplexes Rückwärts-Verschieben.}
\end{factlist}
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def our_hash(word):
    return ord(word[0])           # einfache Hash-Fkt.

def create_hash_table(l, M):
    res = [[] for _ in range(M)]  # M leere Buckets
    for x in l:
        idx = our_hash(x) % M     # % statt mod!
        if x not in res[idx]:     # keine Duplikate
            res[idx].append(x)    # chaining: anhaengen
    return res
\end{lstlisting}
\end{codebox}


\subsection{Quadtrees}\label{sec:09-quadtrees}

\subsubsection{Grundidee}
\begin{defbox}[Quadtree]
Quadtrees sind \ZSFkeyword{volle Quarternärbäume} (jeder Knoten hat entweder $0$ oder exakt $4$ Kinder) zur effizienten Verarbeitung \textbf{räumlicher Daten in 2D} (z.\,B. Rechteck-Abfragen).
\end{defbox}

\subsubsection{Struktur eines Knotens}
\begin{contentbox}
Ein Knoten speichert:
\begin{factlist}
\item untere linke Ecke $l$ und obere rechte Ecke $u$
\item \textbf{Maximalkapazität} $m$
\item Liste $p$ von Punkten
\item $4$ Referenzen auf Kinder-Quadtrees $c_0, c_1, c_2, c_3$
\end{factlist}
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
class QuadTree:
    def __init__(self, l, u, m):
        self.l, self.u = l, u     # untere/obere Ecke
        self.m = m                # Kapazitaet
        self.p = []               # Punkte im Knoten
        self.c = [None]*4         # Kinder c0..c3
\end{lstlisting}
\end{codebox}

\subsubsection{Einfügen \& Split}
\begin{contentbox}
\begin{factlist}
\item Ein Punkt wird zu $p$ hinzugefügt.
\item Überschreitet die Anzahl der Punkte die Kapazität $m$, wird der Knoten in \textbf{vier Viertel} partitioniert.
\item Die Punkte aus $p$ werden auf die Kinder aufgeteilt und das eigene $p$ geleert.
\end{factlist}
\end{contentbox}

\subsubsection{Rechtecks-Abfrage (Range Query)}
\begin{contentbox}
\begin{factlist}
\item Die Suche wächst logarithmisch mit der Anzahl der Punkte: $\mathcal{O}(\log n)$.
\item Schneidet das Such-Rechteck einen Quadranten nicht, wird dieser Zweig \textbf{komplett ignoriert} (\ZSFkeyword{Pruning}).
\end{factlist}
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def query(node, q_l, q_u):
    if not overlaps(node, q_l, q_u):  # Pruning!
        return []                     # Zweig ignorieren
    hits = [pt for pt in node.p       # lokale Punkte
            if inside(pt, q_l, q_u)]
    for child in node.c:              # rekursiv in Kinder
        if child is not None:
            hits += query(child, q_l, q_u)
    return hits
\end{lstlisting}
\end{codebox}

\subsubsection{Bemerkung}
\begin{statementbox}
Die \textbf{Konstruktion} des Baumes ist relativ aufwendig; der Einsatz lohnt sich daher erst bei \textbf{höherfrequentierten Abfragen}.
\end{statementbox}

\begin{tablebox}
\renewcommand{\arraystretch}{1.25}
\begin{ZSFtable}{Q{1.2} L L L}
\toprule
\textbf{Datenstruktur} & \textbf{Suche} & \textbf{Einfügen} & \textbf{Entfernen} \\
\midrule
Liste / Array       & $\mathcal{O}(n)$          & $\mathcal{O}(n)$ \emph{(worst)}   & $\mathcal{O}(n)$ \emph{(worst)}   \\

Heap                & $\mathcal{O}(n)$          & $\mathcal{O}(\log n)$             & $\mathcal{O}(\log n)$\,\footnotesize{(Top)} \\
Hash-Tabelle        & $\mathcal{O}(1)$ \emph{(erw.)} & $\mathcal{O}(1)$ \emph{(erw.)} & $\mathcal{O}(1)$ \emph{(erw.)} \\
Quadtree            & $\mathcal{O}(\log n)$\,\footnotesize{(Range)} & $\mathcal{O}(\log n)$ & --- \\
\bottomrule
\end{ZSFtable}
\end{tablebox}

\begin{statementbox}[Merke]
\begin{factlist}
\ZSFFact{Hash-Tabelle}{schnellster Zugriff im Durchschnitt, aber ungeordnet.}
\ZSFFact{BST / Heap}{geordneter Zugriff, log.\ Laufzeit — nur balanciert effizient.}
\ZSFFact{Quadtree}{spezialisiert auf 2D-Bereichs-Abfragen.}
\end{factlist}
\end{statementbox}
