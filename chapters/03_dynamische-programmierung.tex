\section{Dynamische Programmierung}\label{sec:03-dynamische-programmierung}

% --- Kernidee: DP-Prinzip und zwei Implementierungsansätze ---
\begin{contentbox}[Kernidee]
    Ein komplexes Problem wird in kleinere Teilprobleme zerlegt. Diese werden einmal gelöst und das Ergebnis gespeichert (in einer Liste, Tabelle o.ä.), um es später bei Bedarf wiederzuverwenden.

\begin{factlist}
\ZSFFact{Memoization (Top-down)}{Beginnt mit dem ursprünglichen Problem und löst es rekursiv, speichert aber bereits berechnete Ergebnisse in einem Speicher (z.B. einer Liste oder einem Dictionary). Der Speicher verhindert doppelte Berechnungen.}
\ZSFFact{Dynamische Programmierung (Bottom-up)}{Baut die Lösung iterativ von den kleinsten Teilproblemen auf. Die Ergebnisse werden schrittweise gespeichert und zur Lösung größerer Probleme verwendet. Keine Rekursion nötig.}
\end{factlist}
\end{contentbox}

\begin{statementbox}[Unterschied]
Beide Ansätze nutzen Speicher zur Optimierung, aber:
\begin{factlist}
\ZSFFact{Memoization}{rekursiv + speichert nur tatsächlich verwendete Teilprobleme.}
\ZSFFact{DP (Bottom-up)}{iterativ + berechnet alle möglichen Teilprobleme systematisch.}
\end{factlist}
\end{statementbox}

% Erkennungsmerkmal: Eingabe ist eine Sequenz oder ein Integer n

\subsection{1D-DP: \texttt{dp[i]} — Lineare Sequenzen}\label{sec:03-1d-dp}

% --- Muster-Steckbrief: Zustands- und Übergangsdefinition ---
\begin{contentbox}
\begin{factlist}
\ZSFFact{Zustand}{\texttt{dp[i]} = optimale Lösung für die ersten $i$ Elemente}
\ZSFFact{Übergang}{\texttt{dp[i] = f(dp[i-1], dp[i-2], \ldots)}}
\ZSFFact{Basisfall}{\texttt{dp[0]}, \texttt{dp[1]} explizit setzen}
\ZSFFact{Antwort}{\texttt{dp[n]}}
\ZSFFact{Komplexität}{$O(n)$ Zeit, $O(n)$ Raum}
\end{factlist}
\end{contentbox}

% --- Fibonacci: Bottom-up und Top-down (bestehend) ---
\begin{contentbox}[Fibonacci — Bottom-up]
Iterative Berechnung. Berechnet auch unnötige Elemente, geht in 1D.

\CodeLine{def fib_DP(n):}[gesucht: n-te Fibonacci-Zahl]\\
\CodeLine{F = [None] * (n+1)}[Tabelle anlegen: Indizes 0..n]\\
\OverlineComment{Basisfaelle: F(0) = F(1) = 1}
\CodeLine{F[0] = 1}\\
\CodeLine{F[1] = 1}\\
\OverlineComment{Bottom-Up: Tabelle von klein nach gross befuellen}
\CodeLine{for i in range(2, n+1):}\\
\CodeLine{    F[i] = F[i-1] + F[i-2]}[Rekurrenz anwenden]\\
\CodeLine{return F[n]}[Ergebnis: letzter Eintrag]
\end{contentbox}

\begin{contentbox}[Fibonacci — Top-down]
Rekursive Berechnung mit Memoization. Berechnet nur benötigte Elemente, meistens in 2D. \ZSFref{sec:12-dp-memoization}

\CodeLine{def fibonacci(n, memo = None):}\\
\OverlineComment{memo: Zwischenspeicher fuer bereits berechnete Werte}
\CodeLine{    if memo is None:}\\
\CodeLine{        memo = [1,1] + [None]*(n-1)}\\
\CodeLine{    if memo[n] is not None:}[bereits berechnet -> zurueck]\\
\CodeLine{        return memo[n]}\\
\CodeLine{    else:}[noch nicht berechnet -> rekursiv]\\
\CodeLine{        memo[n-1] = fibonacci(n-1, memo)}\\
\CodeLine{        memo[n-2] = fibonacci(n-2, memo)}\\
\CodeLine{        memo[n] = memo[n-1] + memo[n-2]}[Rekurrenz + speichern]\\
\CodeLine{        return memo[n]}
\end{contentbox}

% --- [EHA] House Robber / Coin Row: 1D-DP mit Auswahlbedingung ---
% [EHA] Exam Hliddal Addition
\begin{contentbox}[{[EHA] House Robber / Coin Row}]
\begin{factlist}
\ZSFFact{Problem}{Array mit Werten; keine zwei direkt benachbarten Elemente gleichzeitig wählbar. Maximiere die Summe.}
\ZSFFact{Übergang}{\texttt{dp[i] = max(dp[i-1],\ dp[i-2] + val[i])}}
\end{factlist}

\CodeLine{def house_robber(nums):}\\
\CodeLine{    n = len(nums)}\\
\CodeLine{    dp = [0] * n}[dp[i]: max. Summe bis Index i]\\
\CodeLine{    dp[0] = nums[0]}[Basisfall: nur erstes Element]\\
\CodeLine{    dp[1] = max(nums[0], nums[1])}[Basisfall: erstes/zweites Element]\\
\OverlineComment{nehmen: aktuell + bestes ohne direkten Nachbar (i-2)}
\OverlineComment{weglassen: bestes Ergebnis bis zum Vorgaenger (i-1)}
\CodeLine{    for i in range(2, n):}\\
\CodeLine{        dp[i] = max(dp[i-1], dp[i-2] + nums[i])}\\
\CodeLine{    return dp[-1]}[Antwort: letzter Eintrag]
\end{contentbox}

\subsection{[EHA] Kapazitäts-DP: \texttt{dp[i][w]} — Knapsack-Familie}\label{sec:03-kapazitaets-dp}

% --- Muster-Steckbrief ---
\begin{contentbox}
\begin{factlist}
\ZSFFact{Zustand}{\texttt{dp[i][w]} = opt. Wert mit ersten $i$ Objekten bei Kapazität $w$}
\ZSFFact{Übergang}{Objekt nehmen oder weglassen}
\ZSFFact{Basisfall}{\texttt{dp[0][*] = dp[*][0] = 0}}
\ZSFFact{Antwort}{\texttt{dp[n][W]}}
\ZSFFact{Komplexität}{$O(n \cdot W)$ Zeit und Raum}
\item \textbf{Varianten:}
    \begin{itemize}
        \item \textbf{0/1 Knapsack} — jedes Objekt max.\ einmal wählbar
        \item \textbf{Unbounded} (Coin Change, Rod Cutting) — Objekte wiederholbar
        \item \textbf{Subset Sum} — Entscheidungsproblem (\texttt{bool})
    \end{itemize}
\end{factlist}
\end{contentbox}

% --- 0/1 Knapsack: jedes Objekt einmal; Übergang schaut auf i-1 ---
\begin{contentbox}[{[EHA] 0/1 Knapsack}]
\begin{factlist}
\ZSFFact{Übergang}{\texttt{dp[i][w] = max(dp[i-1][w],\ val[i] + dp[i-1][w-wt[i]])}}
\end{factlist}
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def knapsack(wt, val, W):
    n = len(val)
    # dp[i][w]: max. Wert mit ersten i Objekten, Kapazitaet w
    dp = [[0]*(W+1) for _ in range(n+1)]
    for i in range(1, n+1):    # ueber alle Objekte iterieren
        for w in range(W+1):
            dp[i][w] = dp[i-1][w]  # Option 1: weglassen
            if wt[i-1] <= w:       # Option 2: nehmen (passt?)
                dp[i][w] = max(dp[i][w],
                    val[i-1] + dp[i-1][w-wt[i-1]])
    return dp[n][W]                # Antwort: rechte untere Ecke
\end{lstlisting}
\end{codebox}

% --- Coin Change: Unbounded-Variante; Übergang schaut auf dp[i][...] (gleiche Zeile) ---
\begin{contentbox}[{[EHA] Coin Change (Unbounded)}]
\begin{factlist}
\ZSFFact{Übergang}{\texttt{dp[a] = min(dp[a],\ dp[a-c] + 1)} für alle Münzen \texttt{c}}
\end{factlist}
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def coin_change(coins, amount):
    # dp[a]: minimale Muenzanzahl fuer Betrag a
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0              # Basisfall: 0 Muenzen fuer Betrag 0
    for a in range(1, amount + 1):  # Betrag aufbauen
        for c in coins:
            if c <= a:     # Muenze c passt noch in Betrag a
                # 1 Muenze c + optimale Loesung fuer Restbetrag
                dp[a] = min(dp[a], dp[a-c] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1
\end{lstlisting}
\end{codebox}

% Übergang kommt von (i-1,j), (i,j-1) oder (i-1,j-1)
% Beispiele: LCS, Edit Distance, Grid Paths, Minimum Path Sum

% [EHA] Exam Hliddal Addition — gesamter Abschnitt neu
\subsection{[EHA] Zwei-Index-DP: \texttt{dp[i][j]} — Gitter \& Strings}\label{sec:03-zwei-index-dp}

% --- Muster-Steckbrief ---
\begin{contentbox}
\begin{factlist}
\ZSFFact{Zustand}{\texttt{dp[i][j]} = opt.\ Lösung für Präfix $i$ von $A$ und Präfix $j$ von $B$}
\ZSFFact{Übergang}{aus \texttt{dp[i-1][j-1]}, \texttt{dp[i-1][j]}, \texttt{dp[i][j-1]}}
\ZSFFact{Basisfall}{erste Zeile \& Spalte (Index 0) explizit setzen}
\ZSFFact{Antwort}{\texttt{dp[m][n]}}
\ZSFFact{Komplexität}{$O(m \cdot n)$ Zeit und Raum}
\end{factlist}
\end{contentbox}

% --- LCS: Übergang diagonal bei Match, sonst Max von links/oben ---
\begin{contentbox}[{[EHA] Longest Common Subsequence (LCS)}]
\textbf{Übergang:}
\begin{factlist}
\ZSFFact{\texttt{X[i]==Y[j]}}{\texttt{dp[i][j] = dp[i-1][j-1] + 1}}
\ZSFFact{sonst}{\texttt{dp[i][j] = max(dp[i-1][j], dp[i][j-1])}}
\end{factlist}
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def lcs(X, Y):
    m, n = len(X), len(Y)
    # dp[i][j]: Laenge der LCS von X[0..i-1] und Y[0..j-1]
    # Basisfall: dp[0][*] = dp[*][0] = 0 (leere Teilfolge)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(1, m+1):
        for j in range(1, n+1):
            if X[i-1] == Y[j-1]:  # Zeichen gleich: diagonal +1
                dp[i][j] = dp[i-1][j-1] + 1
            else:              # ungleich: bestes von oben/links
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    return dp[m][n]                # Antwort: rechte untere Ecke
\end{lstlisting}
\end{codebox}

% --- Grid: Basisfälle erste Zeile/Spalte separat, dann 2D-Übergang ---
\begin{contentbox}[{[EHA] Grid: Minimale Pfadkosten}]
\begin{factlist}
\ZSFFact{Übergang}{\texttt{dp[i][j] = grid[i][j] + min(dp[i-1][j], dp[i][j-1])}}
\end{factlist}
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def min_path(grid):
    m, n = len(grid), len(grid[0])
    # dp[i][j]: minimale Kosten von (0,0) bis Zelle (i,j)
    dp = [[0]*n for _ in range(m)]
    dp[0][0] = grid[0][0]          # Startpunkt
    for j in range(1, n):      # erste Zeile: nur von links
        dp[0][j] = dp[0][j-1] + grid[0][j]
    for i in range(1, m):      # erste Spalte: nur von oben
        dp[i][0] = dp[i-1][0] + grid[i][0]
    for i in range(1, m):
        for j in range(1, n):
            # Zellkosten + bester Vorgaenger (oben/links)
            dp[i][j] = grid[i][j] + min(dp[i-1][j], dp[i][j-1])
    return dp[m-1][n-1]            # Antwort: rechte untere Ecke
\end{lstlisting}
\end{codebox}

% Beispiele: Matrix Chain, Optimal BST, Burst Balloons

% [EHA] Exam Hliddal Addition — gesamter Abschnitt neu
\subsection{[EHA] Intervall-DP: \texttt{dp[i][j]} — Split-Index $k$}\label{sec:03-intervall-dp}

% --- Muster-Steckbrief ---
\begin{contentbox}
\begin{factlist}
\ZSFFact{Zustand}{\texttt{dp[i][j]} = optimale Lösung für Teilproblem über $[i..j]$}
\ZSFFact{Übergang}{probe alle Split-Punkte $k \in [i,\, j{-}1]$: \texttt{dp[i][j] = min\textsubscript{k}\{dp[i][k] + dp[k+1][j] + cost(i,j)\}}}
\ZSFFact{Basisfall}{\texttt{dp[i][i] = 0} (Intervall der Länge 1)}
\ZSFFact{Antwort}{\texttt{dp[0][n-1]}}
\ZSFFact{Komplexität}{$O(n^3)$ Zeit, $O(n^2)$ Raum}
\ZSFFact{Füllreihenfolge}{äussere Schleife über Länge $\ell$, nicht über Zeilen!}
\end{factlist}
\end{contentbox}

% --- Matrix Chain: klassisches Intervall-DP mit Kostenfunktion ---
\begin{contentbox}[{[EHA] Matrix Chain Multiplication}]
Gegeben $n$ Matrizen, Dimensionen via Array $p$ (Länge $n{+}1$).
\begin{factlist}
\ZSFFact{Übergang}{\texttt{dp[i][j] = min\textsubscript{k}\{dp[i][k] + dp[k+1][j] + p[i]·p[k+1]·p[j+1]\}}}
\end{factlist}
\end{contentbox}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
def matrix_chain(p):
    n = len(p) - 1               # Anzahl Matrizen
    # dp[i][j]: min. Mult. fuer Matrizen i bis j
    dp = [[0]*n for _ in range(n)]
    for l in range(2, n+1):      # Intervall-Laenge aufsteigend
        for i in range(n - l + 1):
            j = i + l - 1        # rechtes Ende des Intervalls
            dp[i][j] = float('inf')
            for k in range(i, j):  # alle Split-Punkte probieren
                # linke Gruppe + rechte Gruppe + Kosten Kombination
                cost = (dp[i][k] + dp[k+1][j]
                        + p[i] * p[k+1] * p[j+1])
                dp[i][j] = min(dp[i][j], cost)
    return dp[0][n-1]            # Antwort: gesamtes Intervall
\end{lstlisting}
\end{codebox}
