\label{sec:01-kontrollstrukturen}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]

# von 0 bis 10 in 2er-Schritten
for i in range(0, 10, 2):
    print(i)

for char in "abc":  # über Zeichen iterieren
    print(char)

# Index + Wert
for i, val in enumerate(["a", "b"]):
    print(i, val)

# Dict iterieren
for k, v in {"x": 1, "y": 2}.items():
    print(k, v)
\end{lstlisting}

\subsection{Slicing — Details}
\label{sec:01-slicing-details}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
a = [10,20,30,40,50]        # a[start:stop:step]
# start inklusive, stop exklusiv, Standardschritt=1
a[1:4]    # [20,30,40]      a[:3] / a[2:] / a[:]=copy
a[-1]     # 50              a[-2:]=last 2, a[:-2]=drop 2
a[::2]    # [10,30,50]      a[::-1]=reversed copy
a[4:1:-1] # [50,40,30]      step<0 -> right to left
b = a[:]; b[0]=99           # slice = new list (no alias)
a[1:3] = [0,0,0]            # slice-assign replaces range
a[1:3] = []                 # slice-assign [] deletes
"python"[::-1]              # works on str/tuple (immut.)
\end{lstlisting}
\end{codebox}

\begin{codeboxmid}

\begin{codeboxmid}

\end{codeboxmid}

\subsection{Iteration über ein Dictionary}
\label{sec:01-iteration-ueber-ein-dictionary}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
d = {"x": 1, "y": 2}

for k in d:                # Schlüssel
    print(k)

for v in d.values():       # Werte
    print(v)

for k, v in d.items():     # Schlüssel und Werte
    print(k, v)

# mit Index
for i, (k, v) in enumerate(d.items()):
    print(i, k, v)
\end{lstlisting}
\end{codebox}

\subsection{zip()}
\label{sec:01-zip}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
a = [1, 2, 3]
b = ['a', 'b', 'c']

# Elemente paaren: (1,'a'), (2,'b'), ...
zipped = zip(a, b)
# [(1, 'a'), (2, 'b'), (3, 'c')]
print(list(zipped))

# auch in for-Schleifen verwendbar
for x, y in zip(a, b):
    print(x, y)

\end{lstlisting}
\end{codebox}

% -----------------------------
\subsection{zip() — Entpacken}
\label{sec:01-zip-entpacken}
\begin{codebox}
\begin{lstlisting}[style=CodeExpert]
# gezippte Listen entpacken
a2, b2 = zip(*[(1, 'a'), (2, 'b'), (3, 'c')])
print(a2)                  # (1, 2, 3)
print(b2)                  # ('a', 'b', 'c')

# zip endet beim kürzesten Input
# [(1, 'a'), (2, 'b')]
print(list(zip([1, 2], ['a', 'b', 'c'])))
\end{lstlisting}
\end{codebox}
