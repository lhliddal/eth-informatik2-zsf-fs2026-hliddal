% ============================================================
% \section / \subsection / \subsubsection -> Bar-System
% ============================================================
% Wir ueberschreiben die KOMA-Defaults so, dass sie identisch zu
% \StartChapter / \SubsectionBar arbeiten. So bleiben alle
% bestehenden Kapiteldateien (\section{Titel}, \subsection{Titel})
% unveraendert.

% \section -> ChapterBar mit Kapitel-Palette
\renewcommand{\section}{\@ifstar{\zsf@section@star}{\zsf@section@plain}}
\newcommand{\zsf@section@plain}[1]{%
  \StartChapter{#1}%
}
\newcommand{\zsf@section@star}[1]{%
  \StartFrontChapter{#1}%
}

% \subsection -> SubsectionBar (verwendet die SubsectionBar-Umgebung
% aus 60_boxes; das Spacing-Sequencing wird dort verwaltet).
\renewcommand{\subsection}{\@ifstar{\zsf@subsection@star}{\zsf@subsection@plain}}
\newcommand{\zsf@subsection@plain}[1]{\SubsectionBar{#1}}
\newcommand{\zsf@subsection@star}[1]{\SubsectionBar*{#1}}

% \subsubsection -> kompaktes Inline-Heading in Kapitelfarbe
% (kein Vollbalken, weil im 3-Spalten-Layout sehr eng).
\renewcommand{\subsubsection}{\@ifstar{\zsf@subsubsection@star}{\zsf@subsubsection@plain}}
\newcommand{\zsf@subsubsection@plain}[1]{%
  \par\addvspace{\ZSFspaceS}%
  \noindent\textcolor{\chaptercolor}{\ZSFfontBoxTitle #1}%
  \par\nopagebreak\vspace{\ZSFspaceXS}%
}
\newcommand{\zsf@subsubsection@star}[1]{\zsf@subsubsection@plain{#1}}

\makeatother
