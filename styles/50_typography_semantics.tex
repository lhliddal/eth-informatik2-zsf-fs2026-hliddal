% \ZSFref{label} — kompakter interner Querverweis, gerendert als "(→ 6.2)".
% Verwendung:
%   * wenn eine Stelle ein Konzept/Verfahren nutzt, das in einem ANDEREN
%     Kapitel erklaert wird (z.B. "sortiere via Quicksort").
%   * NICHT fuer Delegationen im selben Kapitel — sparsam einsetzen.
% Ziel-Label via \label{sec:...} direkt nach \section{}/\subsection{} setzen.
% \ref* (hyperref) unterdrueckt den Link, Nummern bleiben automatisch korrekt.
% Farbe = Palette-Farbe des ZIELkapitels (Wegweiser-Prinzip): wird automatisch
% aus der Kapitelnummer der Referenz abgeleitet ("6.6" -> ChapterColor6).
% Aufrecht + fett statt kursiv fuer maximale Lesbarkeit.
\def\ZSF@refExtractFirst#1#2\@nil{#1}
\def\ZSF@refExtractChap#1.#2\@nil{#1}
% Bestimmt \ZSF@refTargetColor aus dem Label; Fallback auf die aktive
% Kapitelfarbe im ersten latexmk-Pass (Label noch undefiniert).
\newcommand{\ZSF@refComputeTargetColor}[1]{%
  \@ifundefined{r@#1}{%
    \edef\ZSF@refTargetColor{\chaptercolor}%
  }{%
    \edef\ZSF@refTargetNum{\expandafter\expandafter\expandafter
      \ZSF@refExtractFirst\csname r@#1\endcsname\@nil}%
    \edef\ZSF@refTargetColor{ChapterColor\expandafter\ZSF@refExtractChap\ZSF@refTargetNum.\@nil}%
  }%
}
\newcommand{\ZSFref}[1]{%
  \begingroup
  \ZSF@refComputeTargetColor{#1}%
  {\ZSFfontNote\upshape\bfseries\color{\ZSF@refTargetColor}($\rightarrow$\,\ref*{#1})}%
  \endgroup
}

}

% ============================================================
% \section / \subsection / \subsubsection -> Bar-System
% ============================================================
% Wir ueberschreiben die KOMA-Defaults so, dass sie identisch zu
% \StartChapter / \SubsectionBar arbeiten. So bleiben alle
% bestehenden Kapiteldateien (\section{Titel}, \subsection{Titel})
% unveraendert.

% \section -> ChapterBar mit Kapitel-Palette
\renewcommand{\section}{\@ifstar{\zsf@section@star}{\zsf@section@plain}}
\newcommand{\zsf@section@plain}[1]{%
  \StartChapter{#1}%
}
\newcommand{\zsf@section@star}[1]{%
  \StartFrontChapter{#1}%
}

% \subsection -> SubsectionBar (verwendet die SubsectionBar-Umgebung
% aus 60_boxes; das Spacing-Sequencing wird dort verwaltet).
\renewcommand{\subsection}{\@ifstar{\zsf@subsection@star}{\zsf@subsection@plain}}
\newcommand{\zsf@subsection@plain}[1]{\SubsectionBar{#1}}
\newcommand{\zsf@subsection@star}[1]{\SubsectionBar*{#1}}

% \subsubsection -> kompaktes Inline-Heading in Kapitelfarbe
% (kein Vollbalken, weil im 3-Spalten-Layout sehr eng).
\renewcommand{\subsubsection}{\@ifstar{\zsf@subsubsection@star}{\zsf@subsubsection@plain}}
\newcommand{\zsf@subsubsection@plain}[1]{%
  \par\addvspace{\ZSFspaceS}%
  \noindent\textcolor{\chaptercolor}{\ZSFfontBoxTitle #1}%
  \par\nopagebreak\vspace{\ZSFspaceXS}%
}
\newcommand{\zsf@subsubsection@star}[1]{\zsf@subsubsection@plain{#1}}

\makeatother
